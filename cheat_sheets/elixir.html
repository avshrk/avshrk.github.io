Elixir

Create a map. 
 
        
          colors = %{ color: "red" }
          cars = %{  "sedan" => "toyota"}
Access map element 
 
        
          colors[:color] 
          cars["sedan"]
Properties of keyword list 
 
        
        keys must be atoms
        keys are ordered
        keys can be duplicated ( utilized in Ecto / maps you can not have duplicate keys)
        Brackets can be omitted if it is the last argument to function
Update a map 
 
        
        colors 
        |> Map.put(:first, "green")

        %{colors | first: "green"}
Add new property to a map 
 
        
        colors
        |> Map.put(:new_prop, new_value)
Create a keyword list 
 
        
        colors = [{:first, "red"}, {:second, "yellow"}]
        colors = [first: "red", second: "green"]
Access the first lement of a keyword list 
 
        
        colors |> hd
        colors |> List.first
        [frst | rest ] = colors

Implement a list length algorithm 
 
        
        defmodule MyList do
          def len([]), do: 0
          def len([_h | tail ]), do: 1 + len(tail)
        end
Implement a function that takes a list of numbers and returns a new list with squared numbers 
 
        
        defmodule Square do
          def square([]), do: []
          def square([h | tail]), do: [h * h | square(tail) ]
        end
